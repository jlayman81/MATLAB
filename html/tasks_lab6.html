task 1

clear all
close all
n0 = 5;
n = 0:2*n0;
h0 = 1/5*sinc((n-n0)/5);
w = [0:(n0-1), n0:-1:0];
%plot(0:length(w)-1,w)
%figure
%plot(n,h0)
h = h0.*w;
%figure
%plot(0:length(h)-1,h);
figure
subplot(1,2,1)
H = fftshift(fft(h));
plot(0:length(H)-1,abs(H))
title('|H(f)|')
subplot(1,2,2)
plot(0:length(H)-1,angle(H))
title('angle(H(f))')
