task2

close all
X = zeros(1,length(n));
for k = 0:N-1
    for ni = 0:N-1
        X(k+1) = X(k+1) + x(ni+1)*exp(-1j*2*pi*k*ni/N);
    end
end
figure
subplot(2,1,1);
plot(n,abs(X));
title('Magnitude')
subplot(2,1,2);
plot(n,angle(X));
title('Phase')
X1 = X;
